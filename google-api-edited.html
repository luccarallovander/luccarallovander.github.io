1. Introduction

I have been charged by the Greater London Authority (GLA), the regional governing body for London’s 33 boroughs, to understand the growing state of dissatisfaction with key public services, namely hospitals due to mounting NHS waiting list times and transport costs and journey times due to Tfl strikes in recent years.

The data compiled in this report will allow the GLA to understand how these two problems interact, by compiling multi-level data. This allows for an analysis of how dissatisfaction in the public service delivery of hospitals can be impacted by borough-level factors such as mean travel times to the nearest hospital, or the mean fare cost to the nearest hospital when using public transport.

2.2 Google Maps Places API Requests

Next, I build a function that queries the Google Maps Places API, one for each of the 33 boroughs in London.

For each hospital I extract: hospital name, business status (i.e. ‘operational’ or ‘not operational’), address, Google rating (0-5), total ratings, longitude and latitude coordinates.

I include eval=FALSE in the following code as the request to the API has a small cost (less than £1 and well within the monthly free allowance). And all results can be reproduced without re-querying the API.

# Set up Google Maps API Key and get Places API URL

key_set("google-maps-api-key")

# Base URL for Places API Text Search
places_url <- "https://maps.googleapis.com/maps/api/place/textsearch/json"
# Function to fetch all results for hospitals in multiple boroughs
fetch_hospitals_in_boroughs <- function(borough_names, keyword) {
  all_hospital_data <- list()
  
  for (borough in borough_names) {
    cat("Fetching hospitals in", borough, "...\n")
    all_results <- list()
    next_page_token <- NULL
    
    repeat {
      # Construct the query
      query <- list(
        query = paste(keyword, "in", borough),
        key = key_get("google-maps-api-key")
      )
      if (!is.null(next_page_token)) {
        query$pagetoken <- next_page_token
        Sys.sleep(2)  # Required delay between paginated requests
      }
      
      # Make the API request
      response <- GET(places_url, query = query)
      data <- content(response, as = "parsed")
      
      # Append results to the list
      all_results <- append(all_results, data$results)
      
      # Check for next_page_token
      next_page_token <- data$next_page_token
      
      # Break the loop if no more pages
      if (is.null(next_page_token)) break
    }
    
    # Extract required fields and convert to tibble
    borough_hospital_data <- lapply(all_results, function(hospital) {
      tibble(
        borough = borough,  # Add the borough as a column
        name = ifelse(!is.null(hospital$name), hospital$name, NA),
        business_status = ifelse(!is.null(hospital$business_status), hospital$business_status, NA),
        formatted_address = ifelse(!is.null(hospital$formatted_address), hospital$formatted_address, NA),
        rating = ifelse(!is.null(hospital$rating), hospital$rating, NA),
        user_ratings_total = ifelse(!is.null(hospital$user_ratings_total), hospital$user_ratings_total, NA),
        lat = ifelse(!is.null(hospital$geometry$location$lat), hospital$geometry$location$lat, NA),
        lng = ifelse(!is.null(hospital$geometry$location$lng), hospital$geometry$location$lng, NA)
      )
    })
    
    # Combine data for this borough
    if (length(borough_hospital_data) > 0) {
      all_hospital_data <- append(all_hospital_data, borough_hospital_data)
    }
  }
  
  # Combine all individual tibbles into one large tibble
  hospital_tibble <- do.call(rbind, all_hospital_data)
  return(hospital_tibble)
}

The data retrieved from the API has its limitations:

These include results that extend beyond the geographical boundaries of the ‘place’ parameter, as Google Maps applies a mix of relevance-based algorithms.

Therefore, the next section cross-references all results to exact borough boundaries, so that the GLA can use the data for precise spatial analyses.

2.3 Georeferencing results to hardline geographic boundaries

I write a function to filter the API results by strict borough boundaries using the shapefile data I scraped.

#Referencing hospitals to hardline geographic boundaries


# First convert shapefile to CRS: 4326 format
gf_transformed <- st_transform(gf, crs = 4326)

# Function to filter hospitals within their respective borough boundaries
filter_hospitals_in_boroughs <- function(hospital_tibble_boroughs, gf_transformed) {
  # Convert hospital data to sf object
  hospital_sf <- st_as_sf(
    hospital_tibble_boroughs,
    coords = c("lng", "lat"),  # Specify longitude and latitude columns
    crs = 4326              
  )
  
  # Initialize an empty list to store filtered hospitals
  filtered_hospitals <- list()
  
  # Loop over each borough
  for (borough in unique(hospital_tibble_boroughs$borough)) {
    # Get the boundary of the current borough
    borough_boundary <- gf_transformed %>% filter(NAME == borough)
    
    # Filter hospitals within the current borough
    borough_hospitals <- hospital_sf %>% filter(borough == !!borough)
    within_borough <- st_within(borough_hospitals, borough_boundary, sparse = FALSE)
    
    # Add only hospitals that fall within the borough boundary
    filtered_hospitals[[borough]] <- borough_hospitals[apply(within_borough, 1, any), ]
  }
  
  # Combine all filtered hospitals into a single sf object
  filtered_hospitals_sf <- do.call(rbind, filtered_hospitals)
  
  return(filtered_hospitals_sf)
}

# Usage
hospitals_within_boroughs <- filter_hospitals_in_boroughs(hospital_tibble_boroughs, gf_transformed)

This filtering takes our total number of hospitals from 1089 down to 507, which shows the importance of cross-referencing Google Maps API results.

The data collected here therefore identifies the (dis)satisfaction and operationality for all hospitals in London Boroughs with precise geographical coordinates.

3. Secondary Data Collection

The secondary data collection focuses on accessibility to the hospitals identified in part 2, measuring on average within boroughs, how easy in terms of time, money and distance it is for residents to travel to their nearest hospital using public transport.

I first generate 20 random point coordinates within each borough to simulate residents’ journey starting points. For each random point, I then find the nearest hospital within that borough using a distance approximation (not API yet). I then query the Google Maps Distance Matrix API for the travel time data between each random point and the nearest hospital, using public transport.

I then aggregate the travel data for each borough to obtain aggregate measures of accessibility.

3.2 Google Maps Distance Matrix API

I now make the requests to the Distance Matrix API between the starting points and nearest hospitals, by specifying the origin and destination coordinates.

get_travel_distance_transit <- function(orig_lat, orig_lon, dest_lat, dest_lon, api_key) {
  # Construct the request URL for Google Maps Distance Matrix API
  origins <- paste0(orig_lat, ',', orig_lon)  # Origin coordinates
  destinations <- paste0(dest_lat, ',', dest_lon)  # Destination coordinates
  
  url <- paste0(
    "https://maps.googleapis.com/maps/api/distancematrix/json?",
    "origins=", origins,
    "&destinations=", destinations,
    "&mode=transit",  # Transit mode
    "&key=", api_key
  )
  
  # Make the API request
  response <- GET(url)
  
  # Parse the response (JSON format)
  response_data <- content(response, "parsed")
  
  # Check if response contains valid elements
  if (!is.null(response_data$rows[[1]]$elements[[1]])) {
    elements <- response_data$rows[[1]]$elements[[1]]
    
    # Extract the relevant fields
    result <- tibble(
      distance = ifelse(!is.null(elements$distance), elements$distance$text, NA),
      duration = ifelse(!is.null(elements$duration), elements$duration$text, NA),
      fare = ifelse(!is.null(elements$fare), elements$fare$text, NA)
    )
    
    return(result)
  } else {
    return(tibble(error = "No valid data returned from API"))
  }
}

Secondly, I store the hospital-level and borough-level data separately in an SQLite database, each in tidy format. This format allows for faster querying and retrieval of specific subsets and is especially useful for multi-level data, given the relational structure.

#Load Data to be stored in SQLite Database

#load the primary data
hospitals_with_lat_lng  <- read_csv(file.path(ddir, "hospitals_with_lat_lng.csv"))

#load the borough-level data select only variables of interest
gf_with_borough_data <- read_csv(file.path(ddir,"gf_with_borough_data.csv"))
borough_level_data <- gf_with_borough_data %>%
  select(NAME, mean_distance_km, mean_duration_mins, mean_fare_gbp, borough_mean_hospital_rating, borough_hospital_count,          borough_operational_count, borough_operational_per)%>%
  rename(borough = NAME)

write_csv(borough_level_data, file.path(ddir, "borough_level_data.csv"))

#Creating database for hospitals and borough data

The example query below shows how both tables can be easily joined using the common key (‘borough’) to access the data.

#Example Query: Find Hospitals with a Rating of 4.5 or Higher and their Borough's Average Hospital Rating.

example_query <- dbGetQuery(hospitals_boroughs_db, "
  SELECT h.name, h.formatted_address, h.rating, b.borough, b.borough_mean_hospital_rating
  FROM hospitals_with_lat_lng AS h
  JOIN borough_level_data AS b
  ON h.borough = b.borough
  WHERE h.rating >= 4.5
")
